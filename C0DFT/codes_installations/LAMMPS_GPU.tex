\subsection{Environment Setup}\label{environment-setup}

\subsubsection{Step 1}\label{step-1}

Download Miniforge installer:

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{wget}\NormalTok{ https://github.com/conda{-}forge/miniforge/releases/latest/download/Miniforge3{-}Linux{-}x86\_64.sh}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 2}\label{step-2}

Run the installer:

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{bash}\NormalTok{ Miniforge3{-}Linux{-}x86\_64.sh}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 3}\label{step-3}

Activate Miniforge:

\begin{Shaded}
\begin{Highlighting}[]
\BuiltInTok{source}\NormalTok{ \textasciitilde{}/miniforge3/bin/activate}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 4}\label{step-4}

Create a new conda environment:

\begin{Shaded}
\begin{Highlighting}[]
\ExtensionTok{conda}\NormalTok{ create }\AttributeTok{{-}n}\NormalTok{ md python=3.11}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 5}\label{step-5}

Activate the environment:

\begin{Shaded}
\begin{Highlighting}[]
\ExtensionTok{conda}\NormalTok{ activate md}
\end{Highlighting}
\end{Shaded}

\subsection{LAMMPS Installation}\label{lammps-installation}

\subsubsection{Step 1}\label{step-1-1}

Install required dependencies:

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{sudo}\NormalTok{ apt update}
\FunctionTok{sudo}\NormalTok{ apt install cmake build{-}essential git libfftw3{-}dev libjpeg{-}dev libpng{-}dev libopenmpi{-}dev}
\end{Highlighting}
\end{Shaded}

Also install CUDA if it's not installed. Check with:

\begin{Shaded}
\begin{Highlighting}[]
\ExtensionTok{nvcc} \AttributeTok{{-}{-}version}
\ExtensionTok{nvidia{-}smi}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 2}\label{step-2-1}

Clone and prepare the LAMMPS repository:

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{git}\NormalTok{ clone https://github.com/lammps/lammps.git}
\BuiltInTok{cd}\NormalTok{ lammps}
\FunctionTok{mkdir}\NormalTok{ build}
\BuiltInTok{cd}\NormalTok{ build}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 3 (Configure)}\label{step-3-configure}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{cmake}\NormalTok{ ../cmake }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ CMAKE\_INSTALL\_PREFIX=../lammps\_gpu }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ BUILD\_MPI=on }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ BUILD\_OMP=on }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ PKG\_KOKKOS=on }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ Kokkos\_ENABLE\_CUDA=on }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ Kokkos\_ARCH\_AMPERE80=on }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ CMAKE\_C\_COMPILER=mpicc }\DataTypeTok{\textbackslash{}}
  \AttributeTok{{-}D}\NormalTok{ CMAKE\_CXX\_COMPILER=mpicxx}
\end{Highlighting}
\end{Shaded}

\subsubsection{Step 4 (Build and
Install)}\label{step-4-build-and-install}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{make} \AttributeTok{{-}j}\VariableTok{$(}\FunctionTok{nproc}\VariableTok{)}
\FunctionTok{make}\NormalTok{ install}
\end{Highlighting}
\end{Shaded}
